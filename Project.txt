CREATE TABLE PERSON(

NAME VARCHAR2(50),

GENDER CHAR(10),

COUNTRY CHAR(20),

CITIES CHAR(20),

BIRTHDATE CHAR(15),

DEATHDATE CHAR(15),

PID CHAR(100)

);



CREATE TABLE ALBUM(

TITLE VARCHAR2(50),

STATUS VARCHAR2(15),

LANGUAGE VARCHAR2(20),

RELEASEDATE VARCHAR2(15),

RID VARCHAR2(100)

);



COMMIT;
